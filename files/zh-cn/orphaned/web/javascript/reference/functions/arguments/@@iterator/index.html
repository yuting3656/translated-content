{{jsSidebar("Functions")}}

@@iterator 属性的初始值是和 {{jsxref("Array.prototype.values")}} 属性的初始值相同的对象。

语法

arguments[Symbol.iterator]()

实例

使用for...of循环的迭代

function f() {
  // 你的浏览器必须支持 for..of 循环
  // 以及 for 循环中的 let 区域变量
  for (let letter of arguments) {
    console.log(letter);
  }
}
f('w', 'y', 'k', 'o', 'p');
